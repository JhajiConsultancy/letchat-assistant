Letchat Widget SDK — One-Line Embed

Option 1 · Floating overlay (one line)

<!-- Paste this single tag before </body> -->
<script
  src="https://cdn.letchat.in/widget.js"
  data-assistant-id="YOUR_ASSISTANT_SLUG"
  async
></script>

Option 2 · Embedded inside a div

<!-- 1. Put a container wherever you want the widget -->
<div id="chat-box" style="width:430px;height:700px"></div>

<!-- 2. Drop the script tag anywhere -->
<script
  src="https://cdn.letchat.in/widget.js"
  data-assistant-id="YOUR_ASSISTANT_SLUG"
  data-mode="embed"
  data-container="#chat-box"
  async
></script>

Option 3 · Programmatic API

<script src="https://cdn.letchat.in/widget.js"></script>
<script>
  LetchatWidget.init({
    assistantId: 'YOUR_ASSISTANT_SLUG',
    mode: 'embed',           // 'full' | 'embed'
    container: '#chat-box',   // CSS selector or HTMLElement
    width:  '430px',
    height: '700px',
    zIndex: 9999,
  });
</script>
